# Fractal Genome Analysis Example

def fractal_genome_analysis(n):

    x = np.linspace(-2, 2, 800)

    y = np.linspace(-2, 2, 800)

    X, Y = np.meshgrid(x, y)

    Z = X + 1j * Y

    C = np.zeros_like(Z, dtype=complex)

    for i in range(n):

        Z = Z**2 + C

        mask = np.abs(Z) > 2

        Z[mask] = 0

        C[mask] = 0

    return np.abs(Z)
